  <Tab title="Video Guides" icon="video">
    Produce explainer videos for onboarding.
    
    ```javascript
    // Generate video via API
    const response = await fetch('https://api.hexus.ai/v1/videos', {
      method: 'POST',
      headers: { 'Authorization': 'Bearer YOUR_API_KEY' },
      body: JSON.stringify({
        prompt: 'Explain login flow for new users',
        duration: 60,
        style: 'professional'
      })
    });
    ```
  </Tab>
  <Tab title="User Guides" icon="book-open">
    Auto-generate documentation from product specs.
    
    <CodeGroup tabs="JavaScript,Python">
      ```javascript
      import { Hexus } from '@hexus/sdk';
      const hexus = new Hexus({ apiKey: 'YOUR_API_KEY' });
      const guide = await hexus.guides.create({
        topic: 'API authentication',
        audience: 'developers'
      });
      ```
      ```python
      from hexus import Hexus
      client = Hexus(api_key="YOUR_API_KEY")
      guide = client.guides.create(
          topic="API authentication",
          audience="developers"
      )
      ```
    </CodeGroup>
  </Tab>
